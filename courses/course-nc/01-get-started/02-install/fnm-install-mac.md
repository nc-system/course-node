    $ curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh

## 2. Install Node into MacOS

    $ curl -fsSL https://fnm.vercel.app/install | bash


## 3. Upgrade

    $ brew upgrade fnm
